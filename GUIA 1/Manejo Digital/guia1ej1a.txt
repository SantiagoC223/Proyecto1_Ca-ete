void setup()
{
  pinMode( 3 , OUTPUT);
}

void loop()
{
  digitalWrite( 6 , LOW);
  digitalWrite( 3 , HIGH);
  delay(5000);
  digitalWrite( 3 , LOW);
  digitalWrite( 6 , HIGH);
  delay(5000);  
  digitalWrite( 6 , LOW);
  digitalWrite( 5 , HIGH);
  delay(5000);
  digitalWrite( 5 , LOW);
  digitalWrite( 6 , HIGH);
  delay(5000);
}
