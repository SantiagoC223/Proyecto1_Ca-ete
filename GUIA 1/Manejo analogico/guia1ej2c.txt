void setup()
{
  pinMode( 2 , OUTPUT);
  pinMode( 3 , OUTPUT);
  pinMode( 4 , OUTPUT);
}

void loop()
{
  analogWrite(2 , 0);
  analogWrite( 3, 255);
  analogWrite( 4, 255);
  delay(1000);
  analogWrite(2 , 255);
  analogWrite( 3, 0);
  analogWrite( 4, 0);
  delay(1000);
  analogWrite( 2 , 255);
  analogWrite( 3, 33);
  analogWrite( 4, 222);
  delay(1000);
  
  
}
