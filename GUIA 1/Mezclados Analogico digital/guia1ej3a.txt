int buzzer = 4;


void setup() {
  
  pinMode( 2 , OUTPUT);
  pinMode( 3 , OUTPUT);
  pinMode( 4 , OUTPUT);
  pinMode( 5 , OUTPUT);
  
  pinMode( buzzer , OUTPUT );
  
  
} 

void loop()
{
  analogWrite( 5 , 0);
  analogWrite( buzzer , 255);
  analogWrite( 2 , 255);
  delay(500);
  analogWrite( 3 , 255);
  delay(500);
  analogWrite( 2 , 0);
  analogWrite( buzzer , 130);
  analogWrite( 5 , 255);
  delay(500);
}
