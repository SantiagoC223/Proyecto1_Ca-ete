int buzzer = 4;

void setup() {
  
  pinMode( 2 , OUTPUT);
  pinMode( 3 , OUTPUT);
  pinMode( 4 , OUTPUT);
  
  pinMode( buzzer , OUTPUT );
  
  
} 

void loop()
{
  delay(5000);
  digitalWrite( 3 , LOW);
  digitalWrite( 2 , HIGH);
  digitalWrite( buzzer , HIGH);
  delay(5000);
  digitalWrite( 2 , LOW);
  digitalWrite( buzzer , LOW);
  digitalWrite( 3 , HIGH);
  delay(5000);

}
