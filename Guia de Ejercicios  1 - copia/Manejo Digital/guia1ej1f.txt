bool B ;

void setup()
{
  pinMode( 2 , INPUT);
  pinMode( 9 , OUTPUT);
}

void loop()
{
  
  B = digitalRead( 2 );
  delay(1);
 
  if ( B == HIGH)
  {
  
    analogWrite( 9 , 255);
    delay(1000);
    analogWrite( 9 , 100 );
    delay(1000);
    
  }else{
  
  	analogWrite( 9 , 0);
  
  }
  
}
